"""
OCEAN 人格情境测评问卷
========================
依据《大五人格.docx》编制：25 道情景式选择题，每维度 5 题。

计分规则：
- 每题 A=4 分，B=3 分，C=2 分，D=1 分。
- 若选项均不符合，用户可选 E 填写开放说明，由 LLM 按《AI 选项评分标准》
  严格版六步法（情境锚定→语义相似度加权平均→关键词强度微调→质量标记→置信度→输出反馈）评分，
  支持 0.5 分精度。
- 各维度 5 题相加，总分范围 5–20 分。

使用方式：
    python 评估与维度分数.py
    python 评估与维度分数.py --demo

依赖：
    pip install openai matplotlib numpy
"""

# 1. 数据模型
# ---------------------------------------------------------------------------

@dataclass
class Option:
    """一道题的选项。"""
    label: str          # A/B/C/D
    text: str           # 选项文本
    score: int          # 该选项对应原始分（A=4，B=3，C=2，D=1）


@dataclass
class Question:
    """一道评估题。"""
    id: int
    text: str                           # 题干（情景描述）
    dimension: str                      # O / C / E / A / N
    options: List[Option]
    open_text_prompt: str               # 当用户选"都不符合"时的引导语


@dataclass
class Answer:
    """用户对一道题的回答。"""
    question_id: int
    is_open_text: bool = False
    choice_score: Optional[float] = None   # 选择题原始得分（1-4，支持 0.5 分）
    open_text: Optional[str] = None        # 开放文本内容
    open_text_score: Optional[float] = None  # LLM 按评分标准给出的最终得分（1-4，支持 0.5 分）
    open_text_reason: Optional[str] = None   # 评分理由（匹配选项、微调、异常值等）
    quality_flag: Optional[str] = None       # valid / weak_match / boundary / off_topic / contradictory
    confidence: Optional[float] = None       # 0.0-1.0

    def final_score(self) -> Optional[float]:
        """返回该题对维度的原始得分（1-4）。"""
        if self.is_open_text:
            return self.open_text_score
        return self.choice_score


@dataclass
class DimensionResult:
    """单个维度的评估结果。"""
    name: str
    total_score: float  # 维度总分 5-20
    avg_score: float    # 维度均分 1-4
    level: str          # 高/中/低
    description: str
    evidence: List[str] = field(default_factory=list)
    quality_note: str = ""  # 数据质量提示


@dataclass
class Report:
    """完整评估报告。"""
    dimension_results: Dict[str, DimensionResult]
    ocean_order: List[str] = field(default_factory=lambda: ["O", "C", "E", "A", "N"])
    open_text_summary: str = ""
    cross_dim_notes: Dict[str, str] = field(default_factory=dict)

    def to_text(self) -> str:
        lines = [
            "=" * 52,
            "OCEAN 人格情境测评报告",
            "=" * 52,
            "计分说明：每题 A=4、B=3、C=2、D=1；开放文本按 AI 评分标准严格版六步法可评 0.5 分精度；每维度 5 题总分 5–20。",
            "",
        ]
        for d in self.ocean_order:
            r = self.dimension_results[d]
            lines.append(f"【{DIMENSION_LABELS[d]} ({d})】")
            lines.append(f"  维度总分：{r.total_score:.2f} / 20  （均分 {r.avg_score:.2f} / 4.0）— {r.level}")
            lines.append(f"  解读：{r.description}")
            if r.quality_note:
                lines.append(f"  质量提示：{r.quality_note}")
            if r.evidence:
                lines.append("  关键证据：")
                for ev in r.evidence:
                    lines.append(f"    - {ev}")
            lines.append("")
        if self.open_text_summary:
            lines.append("开放文本使用情况：")
            lines.append(self.open_text_summary)
            lines.append("")
        if self.cross_dim_notes:
            lines.append("跨题一致性提示：")
            for d, note in self.cross_dim_notes.items():
                lines.append(f"  - {DIMENSION_LABELS[d]}：{note}")
            lines.append("")
        lines.append("=" * 52)
        return "\n".join(lines)

# 3. LLM 评分器（可插拔）
# ---------------------------------------------------------------------------

class LLMScorer:
    """
    按《AI 选项评分标准》严格版六步法对开放文本进行 1–4 分评分（支持 0.5 分精度）。
    默认使用 OpenAI 兼容接口；如果未配置 API key，则回退到命令行交互。

    评分标准六步法：
        1. 情境锚定：锁定题目、四个预设选项及维度行为锚点表。
        2. 语义相似度加权平均：用户文本与 A/B/C/D 选项做语义比对，得到 base_score。
        3. 关键词强度微调：三级强度 × 权重累积，含否定前缀反转与语境方向判定。
        4. 质量标记：valid / weak_match / boundary / off_topic / contradictory。
        5. 置信度：输出 0.0–1.0 的 confidence。
        6. 输出反馈：返回 final_score、quality_flag、confidence、reason。
    """

    # 加分词：行为强度高于基准选项，倾向高分
    INTENSIFY_STRONG = [
        "甚至", "超额", "必须", "立刻", "绝对", "极度享受", "完全无法忍受",
        "毫无疑问", "完全", "极其", "总是", "一定",
    ]
    INTENSIFY_MEDIUM = ["非常", "特别", "格外"]
    INTENSIFY_WEAK = ["很喜欢", "很期待", "很享受", "很感兴趣", "比较喜欢", "倾向于"]

    # 减分词：行为强度低于基准选项，或带有勉强/抗拒情绪，倾向低分
    WEAKEN_STRONG = [
        "抗拒", "讨厌", "不喜欢", "懒得", "完全不", "绝不", "打死也不", "被迫", "不得已",
    ]
    WEAKEN_MEDIUM = [
        "偶尔", "尽量", "看心情", "不得不", "有点", "虽然", "但是", "有时",
        "可能", "大概", "勉强", "不太想", "不情愿",
    ]
    WEAKEN_WEAK = ["不太确定"]

    # 混合词：表达摇摆或情境依赖，出现即触发 ambiguity=true
    MIXED_WORDS = [
        "有时候", "看情况", "一半一半", "视情况而定", "时好时坏", "不一定",
        "摇摆不定", "忽上忽下",
    ]

    def __init__(
        self,
        api_key: Optional[str] = None,
        base_url: Optional[str] = None,
        model: Optional[str] = None,
    ):
        self.api_key = api_key or LLM_API_KEY
        self.base_url = base_url or LLM_BASE_URL
        self.model = model or LLM_MODEL
        self._client = None

    def _get_client(self):
        if self._client is None:
            try:
                import openai
            except ImportError as exc:
                raise ImportError("使用 OpenAI 接口需要安装 openai：pip install openai") from exc
            self._client = openai.OpenAI(api_key=self.api_key, base_url=self.base_url)
        return self._client

    def score_open_text(
        self,
        question: Question,
        user_text: str,
    ) -> Tuple[float, str, str, float]:
        """返回 (1-4 维度最终得分, 评分理由, quality_flag, confidence)。"""
        if not self.api_key or self.api_key in ("YOUR_API_KEY", ""):
            return self._fallback_manual(question, user_text)

        prompt = self._build_grading_prompt(question, user_text)
        system_msg = (
            "你是一位专业的人格心理学评估助手。请严格按照《AI 选项评分标准》严格版六步法"
            "（情境锚定→语义相似度加权平均→关键词强度微调→质量标记→置信度→输出反馈）"
            "对用户的开放回答评分。分数范围 1.0–4.0，支持 0.5 分精度。"
            "必须输出 quality_flag 和 confidence。只输出 JSON 格式，不要输出任何解释性文字。"
        )
        try:
            client = self._get_client()
            resp = client.chat.completions.create(
                model=self.model,
                messages=[
                    {"role": "system", "content": system_msg},
                    {"role": "user", "content": prompt},
                ],
                temperature=0.2,
                response_format={"type": "json_object"},
            )
            content = resp.choices[0].message.content
            data = json.loads(content)
            final_score = float(data.get("final_score", data.get("base_score", 2.5)))
            reason = data.get("reason", "")
            best_match = data.get("best_match", "")
            similarities = data.get("similarities", {})
            quality_flag = data.get("quality_flag", "valid")
            confidence = float(data.get("confidence", 0.7))
            adjustment = float(data.get("adjustment", 0.0))
            detail = self._format_reason(
                best_match, similarities, adjustment, quality_flag, confidence, reason
            )
            return (
                max(1.0, min(4.0, final_score)),
                detail,
                quality_flag,
                max(0.0, min(1.0, confidence)),
            )
        except Exception as exc:
            print(f"[LLM 调用失败] {exc}，转入手动评分。")
            return self._fallback_manual(question, user_text)

    def _build_grading_prompt(self, question: Question, user_text: str) -> str:
        dim_label = DIMENSION_LABELS.get(question.dimension, question.dimension)
        opts = question.options
        anchors = DIMENSION_ANCHORS.get(question.dimension, "")
        return (
            f"=== 第一步：情境锚定 ===\n"
            f"题目编号：第 {question.id} 题\n"
            f"评估维度：{dim_label}\n"
            f"题目情境：{question.text}\n"
            f"四个预设选项及对应原始分：\n"
            f"A（4分）：{opts[0].text}\n"
            f"B（3分）：{opts[1].text}\n"
            f"C（2分）：{opts[2].text}\n"
            f"D（1分）：{opts[3].text}\n\n"
            f"=== 维度行为锚点表（用于 weak_match 或异常值推断） ===\n"
            f"{anchors}\n\n"
            f"=== 第二步：语义相似度计算与 base_score ===\n"
            f"1. 在计算相似度之前，请先忽略选项的排列顺序（A/B/C/D 仅为编号，不代表偏好方向），"
            f"仅基于语义内容进行判断。不要因为 A 排在第一位就倾向给出更高的相似度。\n"
            f"2. 估算用户回答与 A/B/C/D 四个选项的语义相似度 sim_A、sim_B、sim_C、sim_D（0%-100%）。\n"
            f"3. 若四项相似度之和 ≥ 1%，计算加权平均分：\n"
            f"   base_score = (sim_A×4 + sim_B×3 + sim_C×2 + sim_D×1) / (sim_A+sim_B+sim_C+sim_D)\n"
            f"   若四项相似度之和 < 1%，直接判定为 off_topic，final_score = 2.5。\n\n"
            f"相似度门槛：\n"
            f"- 最高相似度 < 35% → weak_match：结合维度行为锚点推断 1.0–4.0 的分数，confidence < 0.5\n"
            f"- 最高 ≥ 80% 且其余 < 20% → strong_match：跳过关键词微调，直接以 weighted_base_score 为 final_score，"
            f"quality_flag=valid，confidence=0.9\n"
            f"- 最高与次高之差 < 10pp → boundary：按竞争规则取中位数或平均值\n"
            f"- 四项相似度两两之差均 < 10pp → ambiguity：base_score = 2.5\n\n"
            f"=== 第三步：关键词强度微调（仅非 strong_match 时执行） ===\n"
            f"加分词-强（weight=1.5）：{', '.join(self.INTENSIFY_STRONG)}\n"
            f"加分词-中（weight=1.0）：{', '.join(self.INTENSIFY_MEDIUM)}\n"
            f"加分词-弱（weight=0.5）：{', '.join(self.INTENSIFY_WEAK)}\n"
            f"减分词-强（weight=1.5）：{', '.join(self.WEAKEN_STRONG)}\n"
            f"减分词-中（weight=1.0）：{', '.join(self.WEAKEN_MEDIUM)}\n"
            f"减分词-弱（weight=0.5）：{', '.join(self.WEAKEN_WEAK)}\n"
            f"混合词（出现即触发 ambiguity=true）：{', '.join(self.MIXED_WORDS)}\n\n"
            f"否定前缀反转规则：检查否定结构（不、没、无、非、别、甭、莫、不太、不是很、"
            f"不怎么、并不、从未、绝不、决不、不要、不能、别去、不准）。"
            f"如果关键词前有否定前缀，反转该关键词的加减方向并降一级权重。"
            f"例如\"不讨厌\"应识别为加分词-弱（weight=0.5）。从最长复合否定开始匹配。双重否定还原为正向。\n\n"
            f"语境方向判定规则：先由 base_score 确定核心行为方向（>2.5 为高分倾向，<2.5 为低分倾向）。"
            f"如果关键词修饰的是低分方向行为，则关键词加减方向反转。"
            f"例如 base_score=1.3，\"绝对不去\"中\"绝对\"强化低分方向 → 从加分词反转为减分词。\n\n"
            f"混合词优先规则：任何混合词匹配 → ambiguity=true，该词不再计入 intensity_boost 或 weaken_penalty。\n\n"
            f"微调计算公式：\n"
            f"intensity_boost = Σ(加分词出现次数 × 该级 weight)  // 先执行否定反转与语境方向判定\n"
            f"weaken_penalty  = Σ(减分词出现次数 × 该级 weight)\n"
            f"net_adjustment  = min(1.5, intensity_boost) - min(1.5, weaken_penalty)\n"
            f"final_score     = base_score + net_adjustment\n"
            f"最终结果限制在 [1.0, 4.0]。\n\n"
            f"=== 第四步：质量标记（quality_flag） ===\n"
            f"| 标记 | 触发条件 |\n"
            f"| valid | 回答与情境相关，语义清晰，最高相似度 ≥ 35% |\n"
            f"| weak_match | 最高相似度 < 35%，但用户确实在认真回答 |\n"
            f"| boundary | 最高与次高相似度之差 < 10pp |\n"
            f"| off_topic | 答非所问、明显敷衍、字数 < 5 个有意义汉字 |\n"
            f"| contradictory | 文本内部自相矛盾，或关键词密度 > 40% |\n\n"
            f"=== 第五步：置信度（confidence） ===\n"
            f"输出 0.0–1.0 的置信度。≥0.7 可信，0.5–0.7 存疑，<0.5 不可信。\n"
            f"置信度调整：strong_match +0.1；boundary -0.1；有效汉字数 ≥15 且关键词密度 <10% +0.1；关键词密度 >40% -0.2。\n"
            f"最终 confidence 限制在 [0.0, 1.0]。\n\n"
            f"=== 用户开放回答 ===\n"
            f"{user_text}\n\n"
            f"=== 输出格式 ===\n"
            f"请只输出如下 JSON（不要 Markdown 代码块）：\n"
            f'{{\n'
            f'  "base_score": float,          // 加权平均后的基础分\n'
            f'  "best_match": "A/B/C/D",      // 最匹配选项\n'
            f'  "similarities": {{"A": float, "B": float, "C": float, "D": float}},  // 0-100\n'
            f'  "final_score": float,         // 最终得分（1.0-4.0，支持0.5）\n'
            f'  "quality_flag": "valid|weak_match|boundary|off_topic|contradictory",\n'
            f'  "confidence": float,          // 0.0-1.0\n'
            f'  "reason": str                 // 简要说明评分依据\n'
            f'}}'
        )

    @staticmethod
    def _format_reason(
        best_match: str,
        similarities: Dict[str, float],
        adjustment: float,
        quality_flag: str,
        confidence: float,
        reason: str,
    ) -> str:
        """把 LLM 返回的结构化信息格式成易读的理由。"""
        parts = []
        if best_match:
            parts.append(f"最匹配选项：{best_match}")
        if similarities and isinstance(similarities, dict):
            sim_str = ", ".join(
                f"{k}:{int(v)}%" for k, v in similarities.items() if isinstance(v, (int, float))
            )
            if sim_str:
                parts.append(f"相似度：{sim_str}")
        if adjustment:
            sign = "+" if adjustment > 0 else ""
            parts.append(f"微调：{sign}{adjustment}")
        parts.append(f"质量标记：{quality_flag}")
        parts.append(f"置信度：{confidence:.2f}")
        if reason:
            parts.append(f"判定：{reason}")
        return "；".join(parts) if parts else "LLM 评分"

    def _fallback_manual(
        self,
        question: Question,
        user_text: str,
    ) -> Tuple[float, str, str, float]:
        """无 LLM 时，让操作者手动输入分数。"""
        print(f"\n[需要手动评分] 维度：{DIMENSION_LABELS.get(question.dimension, question.dimension)}")
        print(f"题目：{question.text}")
        print(f"用户回答：{user_text}")
        while True:
            raw = input("请输入 1-4 分（支持 0.5，如 3.5）：").strip()
            try:
                score = float(raw)
                if 1.0 <= score <= 4.0 and score * 10 % 5 == 0:
                    return score, "手动评分", "valid", 1.0
            except ValueError:
                pass
            print("输入无效，请输入 1.0–4.0 之间且为 0.5 倍数的数字。")

class OCEANEngine:
    def __init__(self, llm_scorer: Optional[LLMScorer] = None):
        self.llm_scorer = llm_scorer or LLMScorer()
        self.answers: Dict[int, Answer] = {}

    def collect_answers(self, input_fn: Callable[[str], str] = input) -> List[Answer]:
        """通过命令行交互收集用户答案。"""
        print("=" * 60)
        print("OCEAN 人格情境测评问卷")
        print("共 25 道题，每题 A-D 四个选项；如都不符合可选 E 填写说明。")
        print("=" * 60)

        answers: List[Answer] = []
        for q in QUESTION_BANK:
            print(f"\n第 {q.id} 题")
            print(q.text)
            for opt in q.options:
                print(f"  {opt.label}. {opt.text}")
            print("  E. 以上都不符合，我想自己说明")

            while True:
                choice = input_fn("请选择（A/B/C/D/E）：").strip().upper()
                if choice in {"A", "B", "C", "D"}:
                    opt_map = {opt.label: opt for opt in q.options}
                    ans = Answer(
                        question_id=q.id,
                        is_open_text=False,
                        choice_score=opt_map[choice].score,
                    )
                    break
                if choice == "E":
                    text = input_fn("请输入你的说明：").strip()
                    score, reason, quality_flag, confidence = self.llm_scorer.score_open_text(q, text)
                    ans = Answer(
                        question_id=q.id,
                        is_open_text=True,
                        open_text=text,
                        open_text_score=score,
                        open_text_reason=reason,
                        quality_flag=quality_flag,
                        confidence=confidence,
                    )
                    print(f"  → 评分：{score}/4，理由：{reason}")
                    break
                print("输入无效，请重新选择。")
            answers.append(ans)

        self.answers = {a.question_id: a for a in answers}
        return answers

    def compute_report(
        self,
        answers: Optional[List[Answer]] = None,
        generate_chart: bool = True,
    ) -> Report:
        """根据答案计算维度分数并生成报告。"""
        if answers is None:
            answers = list(self.answers.values())

        # 汇总到各维度（开放文本 evidence 优先显示）
        dim_scores: Dict[str, List[float]] = {d: [] for d in DIMENSION_LABELS}
        open_evidence: Dict[str, List[str]] = {d: [] for d in DIMENSION_LABELS}
        choice_evidence: Dict[str, List[str]] = {d: [] for d in DIMENSION_LABELS}
        dim_quality_flags: Dict[str, List[str]] = {d: [] for d in DIMENSION_LABELS}
        dim_confidences: Dict[str, List[float]] = {d: [] for d in DIMENSION_LABELS}
        open_text_scores_by_dim: Dict[str, List[float]] = {d: [] for d in DIMENSION_LABELS}

        for ans in answers:
            q = QUESTION_BY_ID[ans.question_id]
            score = ans.final_score()
            if score is None:
                continue
            dim_scores[q.dimension].append(score)
            if ans.is_open_text:
                reason_snippet = ans.open_text_reason or "LLM评分"
                evidence = (
                    f"Q{q.id}: 开放文本“{ans.open_text[:30]}...”→{score}分"
                    f"（{reason_snippet}）"
                )
                open_evidence[q.dimension].append(evidence)
                if ans.quality_flag:
                    dim_quality_flags[q.dimension].append(ans.quality_flag)
                if ans.confidence is not None:
                    dim_confidences[q.dimension].append(ans.confidence)
                open_text_scores_by_dim[q.dimension].append(score)
            else:
                evidence = f"Q{q.id}: 选择{ans.choice_score}分"
                choice_evidence[q.dimension].append(evidence)

        dim_evidence = {
            d: (open_evidence[d] + choice_evidence[d])[:3]
            for d in DIMENSION_LABELS
        }

        # 计算结果
        results: Dict[str, DimensionResult] = {}
        for d in DIMENSION_LABELS:
            scores = dim_scores[d]
            total = sum(scores) if scores else 12.5
            avg = total / len(scores) if scores else 2.5
            level = self._level(avg)

            # 数据质量提示
            quality_note = ""
            non_valid_flags = [f for f in dim_quality_flags[d] if f != "valid"]
            if len(non_valid_flags) >= 2:
                quality_note = "本维度数据质量较低，结果仅供参考。"

            results[d] = DimensionResult(
                name=DIMENSION_LABELS[d],
                total_score=round(total, 2),
                avg_score=round(avg, 2),
                level=level,
                description=DIMENSION_DESCRIPTIONS[d][level],
                evidence=dim_evidence[d][:3],
                quality_note=quality_note,
            )

        # 跨题一致性检测
        cross_dim_notes: Dict[str, str] = {}
        for d in DIMENSION_LABELS:
            open_scores = open_text_scores_by_dim[d]
            if len(open_scores) >= 2:
                score_range = max(open_scores) - min(open_scores)
                if score_range >= 2.0:
                    cross_dim_notes[d] = (
                        "存在较大场景差异；同一人格特质在不同情境下可能有不同表现。"
                    )

        # 开放文本总评
        open_answers = [a for a in answers if a.is_open_text]
        if open_answers:
            open_summary_lines = [
                f"用户有 {len(open_answers)} 道题使用开放说明，"
                f"主要分布在以下维度："
                f"{', '.join(sorted({QUESTION_BY_ID[a.question_id].dimension for a in open_answers}))}。"
                "这些说明已纳入对应维度总分。"
            ]
            for ans in open_answers:
                q = QUESTION_BY_ID[ans.question_id]
                flag = ans.quality_flag or "unknown"
                conf = f"{ans.confidence:.2f}" if ans.confidence is not None else "N/A"
                open_summary_lines.append(
                    f"  Q{q.id} ({q.dimension}): 得分 {ans.open_text_score}, flag={flag}, confidence={conf}"
                )
            open_summary = "\n".join(open_summary_lines)
        else:
            open_summary = "用户全部使用选项作答，无开放文本。"

        report = Report(
            dimension_results=results,
            open_text_summary=open_summary,
            cross_dim_notes=cross_dim_notes,
        )

        if generate_chart:
            self._draw_radar(results)

        return report

    @staticmethod
    def _level(avg: float) -> str:
        """维度均分 1.0–4.0 的等级划分。"""
        if avg >= 3.0:
            return "高"
        if avg >= 2.0:
            return "中"
        return "低"

    @staticmethod
    def _draw_radar(results: Dict[str, DimensionResult], save_path: str = "ocean_radar.png"):
        """绘制雷达图并保存。"""
        try:
            import matplotlib.pyplot as plt
            import numpy as np
        except ImportError:
            print("[提示] 未安装 matplotlib/numpy，跳过雷达图生成。")
            return

        labels = [DIMENSION_LABELS[d] for d in ["O", "C", "E", "A", "N"]]
        values = [results[d].total_score for d in ["O", "C", "E", "A", "N"]]
        values += values[:1]  # 闭合

        angles = np.linspace(0, 2 * np.pi, len(labels), endpoint=False).tolist()
        angles += angles[:1]

        fig, ax = plt.subplots(figsize=(6, 6), subplot_kw=dict(polar=True))
        ax.plot(angles, values, "o-", linewidth=2, color="#2E86AB")
        ax.fill(angles, values, alpha=0.25, color="#2E86AB")
        ax.set_xticks(angles[:-1])
        ax.set_xticklabels(labels, fontsize=10)
        ax.set_ylim(5, 20)
        ax.set_title("OCEAN 人格维度雷达图", fontsize=14, pad=20)
        plt.tight_layout()
        plt.savefig(save_path, dpi=150)
        print(f"[雷达图已保存] {save_path}")
